## Overview

You can write your own inhibitors by extending the `Inhibitor` class. Custom inhibitors work exactly the same as built-in ones and are registered the same way.

## Creating an inhibitor
```ts src/inhibitors/ServerOnly.ts
import { Inhibitor, ok, err } from "@flint.js/core"
import type { FlintClient } from "@flint.js/core"
import type { BaseCommand, CommandContext } from "@flint.js/core"

export default class ServerOnly extends Inhibitor {
    constructor() {
        super("serverOnly")
    }

    async run(command: BaseCommand, ctx: CommandContext) {
        if (!ctx.message.guild) {
            return err("unknown", { message: "This command can only be used in a server." })
        }

        return ok()
    }
}
```

Then register it on your `InhibitorHandler`:
```ts
this.inhibitorHandler = new InhibitorHandler(this, {
    directory: "./src/inhibitors",
    builtins: [
        new Disabled(),
        new ChannelType(),
        new UserPermissions(),
        new BotPermissions(),
        new ServerOnly()
    ]
})
```

Or load it from your inhibitors directory automatically via `loadAll()` by placing it in `src/inhibitors/`.

## Returning results

Use the `ok` and `err` helpers to return results from your inhibitor:
```ts
import { ok, err } from "@flint.js/core"

// allow the command to run
return ok()

// block the command
return err("unknown")
```

The `err` function accepts a reason and an optional extra data object:
```ts
// with a custom reason
return err("unknown")

// with extra data
return err("cooldown", { remaining: 3000, formatted: "3s" })

// with permissions data
return err("user.permissions", { missing: ["BanMembers"] })
```

## Inhibitor results reference

The full list of reasons and their extra fields:

| Reason | Extra fields | Description |
|---|---|---|
| `"disabled"` | None | Command is disabled |
| `"channel"` | None | Command used in wrong channel |
| `"user.permissions"` | `missing?: PermissionResolvable[]` | User is missing permissions |
| `"bot.permissions"` | `missing?: PermissionResolvable[]` | Bot is missing permissions |
| `"cooldown"` | `remaining?: number`, `formatted?: string` | Command is on cooldown |
| `"unknown"` | None | Catch-all for custom inhibitors |

## Adding a custom reason

If none of the built-in reasons fit your use case, use `"unknown"` as the reason and handle it by name in your `CommandDenied` listener:
```ts src/listeners/commandDenied.ts
import { defineListener, FlintClientEvents } from "@flint.js/core"

export default defineListener({
    name: "commandDenied",
    event: FlintClientEvents.CommandDenied,

    async execute(client, { result, ctx }) {
        switch (result.reason) {
            case "disabled":
                return ctx.message.reply("That command is currently disabled.")
            case "channel":
                return ctx.message.reply("That command cannot be used in this channel.")
            case "user.permissions":
                return ctx.message.reply(`You are missing: ${result.missing?.join(", ")}`)
            case "bot.permissions":
                return ctx.message.reply(`I am missing: ${result.missing?.join(", ")}`)
            case "cooldown":
                return ctx.message.reply(`Please wait ${result.formatted} before using that command again.`)
            case "unknown":
                return ctx.message.reply("You cannot use that command.")
        }
    }
})
```

## Full example

A cooldown inhibitor that limits how often a user can run a command:
```ts src/inhibitors/Cooldown.ts
import { Inhibitor, ok, err } from "@flint.js/core"
import type { BaseCommand, CommandContext } from "@flint.js/core"

export default class Cooldown extends Inhibitor {

    #cooldowns = new Map<string, number>()

    constructor() {
        super("cooldown")
    }

    async run(command: BaseCommand, ctx: CommandContext) {
        if (!command.cooldown) return ok()

        const key = `${ctx.message.author.id}:${command.name}`
        const last = this.#cooldowns.get(key)
        const now = Date.now()

        if (last && now - last < command.cooldown) {
            const remaining = command.cooldown - (now - last)
            return err("cooldown", {
                remaining,
                formatted: `${(remaining / 1000).toFixed(1)}s`
            })
        }

        this.#cooldowns.set(key, now)
        return ok()
    }
}
```
