## Creating your client

Start by creating a client class that extends `FlintClient`:
```ts src/index.ts
import "dotenv/config"
import { FlintClient, CommandHandler, ListenerHandler } from "@flint.js/core"

class MyClient extends FlintClient {
    commandHandler: CommandHandler
    listenerHandler: ListenerHandler

    constructor() {
        super()

        this.listenerHandler = new ListenerHandler(this, {
            directory: "./src/listeners"
        })

        this.commandHandler = new CommandHandler(this, {
            directory: "./src/commands",
            prefix: "!"
        })

        this.commandHandler
            .useListenerHandler(this.listenerHandler)
    }

    override async login(token: string) {
        await this.listenerHandler.loadAll()
        await this.commandHandler.loadAll()
        return super.login(token)
    }
}

const client = new MyClient()
await client.login(process.env.TOKEN as string)
```

<Warning>
    Make sure to loadAll `listenerHandler`, `inhibitorHandler`, `monitorHandler`, and `commandHandler` in the login function.
    `loadAll` is an async function so it needs to be `await`ed, otherwise they wont work properly.
</Warning>

## Creating a command

Create your first command using `defineCommand`:
```ts src/commands/ping.ts
import { defineCommand } from "@flint.js/core"

export default defineCommand({
    name: "ping",
    description: "Pong!",

    async execute(client, message) {
        await message.reply("Pong! 🏓")
    }
})
```

## Creating a listener

Create a listener for the `ready` event:
```ts src/listeners/ready.ts
import { defineListener } from "@flint.js/core"
import { Events } from "@fluxerjs/core"

export default defineListener({
    name: "ready",
    event: Events.Ready,
    once: true,

    async execute(client) {
        console.log(`Logged in as ${client.user?.username}!`)
    }
})
```

## Running your bot

Compile and run your bot:

<CodeGroup>
```bash npm
npx tsc && node dist/index.js
```
```bash pnpm
pnpm tsc && node dist/index.js
```
</CodeGroup>

Your bot is now online and will respond to `!ping` with `Pong! 🏓`.
