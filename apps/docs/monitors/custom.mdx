## Overview

You can write your own monitors by extending the `Monitor` class. Each monitor receives every message that passes its filter options and can do anything it needs with it.

## Creating a monitor
```ts src/monitors/Logger.ts
import { Monitor } from "@flint.js/core"
import type { FlintClient } from "@flint.js/core"
import type { Message } from "@fluxerjs/core"

export default class MessageLogger extends Monitor {
    constructor() {
        super("messageLogger", {
            allowBots: false,
            allowDMs: true
        })
    }

    async run(client: FlintClient, message: Message) {
        console.log(`[${message.author.username}] ${message.content}`)
    }
}
```

Place the file in your monitors directory and it will be loaded automatically by `MonitorHandler`.

## Constructor options

| Option | Type | Default | Description |
|---|---|---|---|
| `allowBots` | `boolean` | `false` | Whether to run on messages from bots |
| `allowDMs` | `boolean` | `false` | Whether to run on direct messages |
| `disabled` | `boolean` | `false` | Whether the monitor is disabled |

## Configurable monitors

Pass options through the constructor to make your monitor configurable:
```ts src/monitors/SpamFilter.ts
import { Monitor } from "@flint.js/core"
import type { FlintClient } from "@flint.js/core"
import type { Message } from "@fluxerjs/core"

export interface SpamFilterOptions {
    threshold?: number
    interval?: number
    action?: "delete" | "warn" | "both"
}

export default class SpamFilter extends Monitor {

    #messageCount = new Map<string, number>()
    #threshold: number
    #interval: number
    #action: "delete" | "warn" | "both"

    constructor(options: SpamFilterOptions = {}) {
        super("spamFilter", {
            allowBots: false,
            allowDMs: false
        })

        this.#threshold = options.threshold ?? 5
        this.#interval = options.interval ?? 5000
        this.#action = options.action ?? "both"
    }

    async run(client: FlintClient, message: Message) {
        const count = (this.#messageCount.get(message.author.id) ?? 0) + 1
        this.#messageCount.set(message.author.id, count)

        setTimeout(() => {
            this.#messageCount.delete(message.author.id)
        }, this.#interval)

        if (count >= this.#threshold) {
            if (this.#action === "delete" || this.#action === "both") {
                await message.delete().catch(() => null)
            }
            if (this.#action === "warn" || this.#action === "both") {
                await message.channel.send(`${message.author.toString()}, please slow down!`)
            }
        }
    }
}
```

Then register it with your options:
```ts
this.monitorHandler = new MonitorHandler(this, {
    directory: "./src/monitors",
    builtins: [
        new SpamFilter({ threshold: 3, interval: 3000, action: "both" })
    ]
})
```

## Full example

A monitor that filters messages containing invite links:
```ts src/monitors/InviteFilter.ts
import { Monitor } from "@flint.js/core"
import type { FlintClient } from "@flint.js/core"
import type { Message } from "@fluxerjs/core"

const INVITE_REGEX = /(?:web\.)?(?:fluxer\.gg|fluxer\.app\/invite|fluxer\.com\/invite)\/[a-zA-Z0-9]+/gim

export default class InviteFilter extends Monitor {
    constructor() {
        super("inviteFilter", {
            allowBots: false,
            allowDMs: false
        })
    }

    async run(client: FlintClient, message: Message) {
        if (!INVITE_REGEX.test(message.content)) return

        await message.delete().catch(() => null)
        await message.channel.send(
            `${message.author.toString()}, posting invite links is not allowed.`
        )
    }
}
```
